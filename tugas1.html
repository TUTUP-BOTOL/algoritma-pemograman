# Aplikasi Laundry Kasir

print("===================================")
print("        APLIKASI LAUNDRY")
print("===================================")

while True:

    print("\n====== MENU PAKET LAUNDRY ======")
    print("1. Cuci Kering   : Rp 5000/kg")
    print("2. Cuci Setrika  : Rp 7000/kg")
    print("3. Setrika Saja  : Rp 4000/kg")

    # Input data pelanggan
    nama = input("\nNama pelanggan   : ")
    berat = float(input("Berat cucian (kg): "))

    # paket
    pilihan = input("Pilih paket (1/2/3): ")

    # harga paket
    if pilihan == "1":
        paket = "Cuci Kering"
        harga = 5000

    elif pilihan == "2":
        paket = "Cuci Setrika"
        harga = 7000

    elif pilihan == "3":
        paket = "Setrika Saja"
        harga = 4000

    else:
        print("Pilihan tidak tersedia!")
        continue

    # harga
    total = berat * harga

    # Cetak struk
    print("\n========== STRUK LAUNDRY ==========")
    print("Nama Pelanggan :", nama)
    print("Paket Laundry  :", paket)
    print("Berat Cucian   :", berat, "kg")
    print("Harga per Kg   : Rp", harga)
    print("-----------------------------------")
    print("Total Bayar    : Rp", total)
    print("===================================")
    print("Terima kasih telah menggunakan")
    print("jasa laundry kami")
    print("===================================")

    # Input pelanggan lagi?
    lagi = input("\nInput pelanggan lagi? (y/t): ")

    if lagi.lower() == "t":
        print("\nProgram selesai")
        break
